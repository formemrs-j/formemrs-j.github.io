js代码

js代码主要是生成二维码图片的，和倒计时的逻辑，都是qrcode.js实现的


$(document).ready(function() {

    $(window).focus(function() {
        document.location.reload();
    });

    var validTimeSeconds = 5 * 60;
    var timerValidTime = null;

    if (remainingSeconds !== 0) {

        $(".remaining-time").text(parseInt(remainingSeconds / 60) + "分" + (remainingSeconds - 60 * parseInt(remainingSeconds / 60)) + "秒");

        var timer = setInterval(function(){
            remainingSeconds -= 1;
            $(".remaining-time").text(parseInt(remainingSeconds / 60) + "分" + (remainingSeconds - 60 * parseInt(remainingSeconds / 60)) + "秒");
            if (remainingSeconds === 0) {
                clearInterval(timer);
                $(".remaining-container").hide();
                $(".content-div").removeClass("content-animate");
            }
        }, 1000);


    } else {
        $(".valid-time").text(parseInt(validTimeSeconds / 60) + "分" + (validTimeSeconds - 60 * parseInt(validTimeSeconds / 60)) + "秒");

        timerValidTime = setInterval(function(){
            validTimeSeconds -= 1;
            $(".valid-time").text(parseInt(validTimeSeconds / 60) + "分" + (validTimeSeconds - 60 * parseInt(validTimeSeconds / 60)) + "秒");
            if (validTimeSeconds === 0) {
                clearInterval(timerValidTime);
                timerValidTime = null;
                $(".valid-time").text("已过期！");
            }
        }, 1000);
    }

    $(".text-hz-qrcode").css("color", color);
    QrCodeWithLogo.toCanvas({
        canvas: document.getElementById('canvas'),
        content: url + statusUrl,
        width: 200,
        nodeQrCodeOptions: {
          margin: 2,
          errorCorrectionLevel: 'H',
          color: {
            dark: color,  // Blue dots
            light: '#fff' // Transparent background
          }
        }
    });

    $("#canvas").click(function() {
        var w = Math.round(new Date() / 1000);
		var j = a.toString();
        $.get("https://nco.zjgsu.edu.cn/checkin/2137010101?timestamp="+j, function(result) {
            if (result.success) {
                QrCodeWithLogo.toCanvas({
                    canvas: document.getElementById('canvas'),
                    content:  result.url + statusUrl,
                    width: 200,

                    nodeQrCodeOptions: {
                      margin: 2,
                      errorCorrectionLevel: 'H',
                      color: {
                        dark: color,  // Blue dots
                        light: '#fff' // Transparent background
                      }
                    }
                });
                validTimeSeconds = 5 * 60;
                if ($(".valid-time").length > 0) {
                    $(".valid-time").text(parseInt(validTimeSeconds / 60) + "分" + (validTimeSeconds - 60 * parseInt(validTimeSeconds / 60)) + "秒");
                    if (timerValidTime === null) {
                        timerValidTime = setInterval(function(){
                            validTimeSeconds -= 1;
                            $(".valid-time").text(parseInt(validTimeSeconds / 60) + "分" + (validTimeSeconds - 60 * parseInt(validTimeSeconds / 60)) + "秒");
                            if (validTimeSeconds === 0) {
                                clearInterval(timerValidTime);
                                timerValidTime = null;
                                $(".valid-time").text("已过期！");
                            }
                        }, 1000);
                    }
                }
            } else{
                alert('刷新成功！');
            }
        });

    });
});
